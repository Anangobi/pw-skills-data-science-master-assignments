class gobi:
    def prime(n):
        m=[]
        
        for i in range (1,n):
            choice=1
            for j in range(2,i):
                if i%j==0:
                    choice=0
                    break
                else:
                    choice=1
            if choice==1:
                m.append(i)
        return m
